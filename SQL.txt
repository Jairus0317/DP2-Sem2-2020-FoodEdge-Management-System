DROP TABLE IF EXISTS menu;

CREATE TABLE menu(
	ID INT PRIMARY KEY AUTO_INCREMENT,
	Name VARCHAR(50) NOT NULL,
	Items VARCHAR(1020) NOT NULL, 
    	Price DOUBLE UNSIGNED NOT NULL,
	Quantity INT UNSIGNED DEFAULT 1,
	Image VARCHAR(50),
	Category VARCHAR(50) NOT NULL
);
